   15 -- | Standardgebühr
   16 fee :: Int
   17 fee = 1
   18 
   19 -- | Zieht die Standardgebühr von einem gegebenen Chipstapel ab.
   20 --
   21 -- Ergibt keine negative Werte, stattdessen 0.
   22 -- 
   23 -- Beispiele:
   24 --
   25 -- >>> charge 10
   26 -- 9
   27 --
   28 -- >>> charge 0
   29 -- 0
   30 --
   31 charge :: Int -> Int
   32 charge a = if a > fee then a - fee else 0
   33 
   34 -- | Addiert zwei Werte zusammen und zieht die Standardgebühr ab.
   35 --
   36 -- Ergibt keine negative Werte, stattdessen 0.
   37 --
   38 -- Beispiele:
   39 --
   40 -- >>> putChips 4 2
   41 -- 5
   42 --
   43 -- >>> putChips 0 0
   44 -- 0
   45 --
   46 putChips :: Int -> Int -> Int
   47 putChips owned bought = charge (owned + bought)
   48 
   49 -- | @takeChips x y@ nimmt @y@ Chips von der Chipstabel @x@.
   50 --
   51 -- Ergibt keine negative Werte, stattdessen 0.
   52 -- 
   53 -- Beispiele:
   54 --
   55 -- >>> takeChips 4 2
   56 -- 2
   57 --
   58 -- >>> takeChips 2 4
   59 -- 0
   60 --
   61 takeChips :: Int -> Int -> Int
   62 takeChips owned taken = let result = owned - taken in 

   63             if result >= 0 then result else 0
   64 
   65 -- | Berechnet einen Gewinn, bei dem zwei Werten multipliziert werden
   66 -- und eine gestaffelte Gebühr abgezogen wird.
   67 --
   68 -- Beispiele:
   69 --
   70 -- >>> win 20 0
   71 -- 0
   72 --
   73 -- >>> win 20 5
   74 -- 72
   75 --
   76 -- >>> win 5 20
   77 -- 72
   78 --
   79 -- >>> win 20 2
   80 -- 12
   81 --
   82 win :: Int -> Int -> Int
   83 win a b = if a > b then recurse a b 0 else recurse b a 0

   84     where

   85     recurse _ 0 p = p

   86     recurse 0 _ p = p

   87     recurse a b p = 

   88         let b' = if a `mod` 10 == 0 then b-1 else b 

   89         in recurse (a - 1) b' $! (b + p)

   91 data Command
   92     = Put Command Command   
   93     -- ^ Addiert zwei Werte zusammen und zieht die Standardgebühr ab
   94     | Take Command Command  
   95     -- ^ Subtrahiert das zweite Wert vom Ersten
   96     | Win Command Command 
   97     -- ^ Berechnet einen Gewinn, bei dem zwei Werten mutpliziert werden
   98     -- und eine gestaffelte Gebühr abgezogen wird
   99     | Val Int
  100     -- ^ Konstanter Wert
  101     | Sig Command
  102     -- ^ Ergibt 1 für positive Werte, sonst 0
  103 
  104 -- | Konstanter Wert (für negative Zahlen nicht definiert!)
  105 toCommand :: Int -> Command
  106 toCommand = fromIntegral

  108 -- | Evaluiert den gegebenen Command
  109 eval :: Command -> Int
  110 eval (Put c1 c2)  = putChips (eval c1) (eval c2)

  111 eval (Take c1 c2) = takeChips (eval c1) (eval c2)

  112 eval (Win c1 c2)  = win (eval c1) (eval c2)

  113 eval (Val v)      = v

  114 eval (Sig v)      = signum (eval v)

  117 -- | Konvertiert einen Command in einen mathematischen Ausdruck
  118 toString :: Command -> String
  119 toString = additive

  120     -- Hier könnte man genauso gut die einfache Variante nehmen,

  121     -- die alles mit Klammern ausgibt.

  122     where

  123     additive (Put x y)  = additive x ++ " + " ++ multiplicative y

  124     additive (Take x y) = additive x ++ " - " ++ multiplicative y

  125     additive command    = multiplicative command

  126 

  127     multiplicative (Win x y) = multiplicative x ++ " * " ++ atomic y

  128     multiplicative command = unary command

  129 

  130     unary (Sig x) = "signum " ++ atomic x

  131     unary command = atomic command

  132 

  133     atomic (Val n) = show n

  134     atomic command = "(" ++ additive command ++ ")"

  136 -- | Macht die letzte Operation rückgängig, falls die einen 'Take' mit einem 'Val'
  137 -- als zweiten Operand war. Ansonsten bleibt der Wert unverändert.
  138 --
  139 -- Beispiel:
  140 -- >>> payback [Val 0, Val 1, Take (Val 5) (Val 10), Win (Val 4) (Val 2)]
  141 -- [0,1,5,4 * 2]
  142 --
  143 payback :: [Command] -> [Command]
  144 payback [] = []

  145 payback ((Take op1 (Val _)):xs) = op1: payback xs

  146 payback (x:xs) = x:payback xs
  147 
  148 -- | Listet alle Blattwerte eines Commands.
  149 values (Val v) = [v]

  150 values (Put a b) = (values a) ++ (values b)

  151 values (Take a b) = (values a) ++ (values b)

  152 values (Win a b) = (values a) ++ (values b)
  153 
  154 -- | Listet alle Blattwerte einer Liste von Commands
  155 listValues = (foldr (++) []) . (map values)
  156 
  157 
  158 -- | Zieht alle Einsätze der gegebenen Liste von der gegebenen Stapel ab.
  159 --
  160 -- >>> badluck (Val 10) [Val 4, Val 2]
  161 -- 10 - 4 - 2
  162 --
  163 badluck :: Command -> [Command] -> Command
  164 badluck start bets = foldl Take start bets

  169 -- | Commands werden nur nach Wert verglichen, die Struktur wird ignoriert
  170 instance Eq Command where
  171     c1 == c2 = eval c1 == eval c2
  172 
  173 -- | Commands werden nur nach Wert verglichen, die Struktur wird ignoriert
  174 instance Ord Command where
  175     c1 <= c2 = eval c1 <= eval c2
  176     
  177 -- | 'fromInteger' ist für negative Zahlen nicht definiert.
  178 instance Num Command where
  179     c1 + c2 = Put c1 c2
  180     c1 - c2 = Take c1 c2
  181     c1 * c2 = Win c1 c2
  182     abs c1 = c1
  183     signum c1 = c1
  184     fromInteger v

  185         | v >= 0 = Val (fromInteger v)
